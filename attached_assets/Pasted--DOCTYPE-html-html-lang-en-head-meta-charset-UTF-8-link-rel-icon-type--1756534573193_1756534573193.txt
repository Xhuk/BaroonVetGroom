    <script>
        // 5. Your JavaScript code to initialize the map
        
        // The L variable is now available globally because we included the Leaflet JS CDN.

        // Initialize the map and set its view to our chosen geographical coordinates and a zoom level
        // Coordinates for Zócalo, Mexico City
        const map = L.map('map').setView([19.4326, -99.1332], 14);

        // Add a tile layer to the map. This is the background imagery.
        // We're using OpenStreetMap, a free and open-source map.
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 19,
            attribution: '&copy; <a href="http://www.openstreetmap.org/copyright">OpenStreetMap</a>'
        }).addTo(map);

        // Add a marker to the map at a specific location
        const marker = L.marker([19.4326, -99.1332]).addTo(map);

        // Add a popup to the marker that appears when clicked
        marker.bindPopup("<b>¡Hola, México!</b><br>Esto es el Zócalo.").openPopup();

        // You can also add other shapes, like a circle
        const circle = L.circle([19.4352, -99.1412], { // Palacio de Bellas Artes
            color: 'blue',
            fillColor: '#30f',
            fillOpacity: 0.2,
            radius: 500 // Radius in meters
        }).addTo(map);
        
        circle.bindPopup("Palacio de Bellas Artes.");

        console.log("Leaflet map initialized successfully!");
    </script>
